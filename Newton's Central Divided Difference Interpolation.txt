/*
Write a program to approximate the functional value at any given x from given n no. of data using Newtonâ€™s central divided difference interpolation.
*/
#include <stdio.h>

int main() {
    int n, i, j;
    float x[10], y[10][10], xp, yp, h, p;

    printf("Enter the number of data points: ");
    scanf("%d", &n);

    printf("Enter the x and y values:\n");
    for (i = 0; i < n; i++) {
        scanf("%f %f", &x[i], &y[i][0]);
    }

    for (j = 1; j < n; j++) {
        for (i = 0; i < n - j; i++) {
            y[i][j] = (y[i + 1][j - 1] - y[i][j - 1]) / (x[i + j] - x[i]);
        }
    }

    printf("Enter the value of x to approximate: ");
    scanf("%f", &xp);

    yp = y[0][0];
    for (i = 1; i < n; i++) {
        float term = y[0][i];
        for (j = 0; j < i; j++) {
            term *= (xp - x[j]);
        }
        yp += term;
    }

    printf("Approximated value at x = %.2f is y = %.4f\n", xp, yp);
    return 0;
}
